  pro.formateTime = function (template){
		template = template || '{0}年{1}月{2}日 {3}时{4}分{5}秒';
 		var ary = this.match(/\d+/g);
  	template =  template.replace(/\{(\d+)\}/g, function(){
    var n = arguments[1],
    val = ary[n] || '0';
    val < 10 ? val ='0' + val : null;
    return val;
  });
  return template;  
  }
